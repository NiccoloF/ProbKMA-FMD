101 const arma::ivec& s_k = S_k.col(index);
102
103 // new vec with zero at the top
104 arma::ivec len_elong_k_zero(len_elong_k.size() + 1, arma::fill::zeros);
105 std::copy(len_elong_k.begin(), len_elong_k.end(), len_elong_k_zero.begin() + 1);
106
107 // create a matrix whose column_i contains the vector s_k - len_elong_k_zero[i]
108 arma::uword len_elong_k_zero_size = len_elong_k_zero.size();
109 arma::imat s_k_elong_left_right_temp(s_k.n_elem, len_elong_k_zero_size);
110
111 for (arma::uword i=0; i < len_elong_k_zero_size;++i) {
112 s_k_elong_left_right_temp.col(i) = s_k - len_elong_k_zero(i);
113 }
114
115 // create a sequence of integer from len_elong_k_zero.size() to 1
116 arma::ivec reversedSequence = arma::regspace<arma::ivec>(len_elong_k_zero_size,-1,1);
117 reversedSequence(0) -= 1;
118
119 // repeat each col of s_k_elong_left_right a number of times specified by reversedSequence
120 std::vector<arma::ivec> s_k_elong_left_right = util::repeat_elements(s_k_elong_left_right_temp, reversedSequence);
121
122 std::vector<arma::ivec> len_elong_k_right_list(len_elong_k_zero_size);
123 const int max_len_elong_k = len_elong_k.back();
124 unsigned int v_dom_elong_size = 0;
125
126 for (unsigned int i = 0; i < len_elong_k_zero_size; ++i) {
127 len_elong_k_right_list[i] = len_elong_k_zero.elem(find(len_elong_k_zero <= max_len_elong_k - len_elong_k_zero(i)));
128 v_dom_elong_size += len_elong_k_right_list[i].size();
129 }
130
131 // v_dom_elong_left_right will be a vector of arma::uvec containing all the elongated version of v_dom_k
132 std::vector<arma::urowvec> v_dom_elong_left_right(v_dom_elong_size);
133 const int v_dom_k_len = v_dom_k.n_elem;
134 unsigned int k = 0;
135 for (unsigned int i = 0; i < len_elong_k_zero_size; ++i){
136 const arma::ivec & leng_elong_right_vector = len_elong_k_right_list[i];
137 const unsigned int leng_elong_left = len_elong_k_zero(i);
138 for (unsigned int j = 0; j < leng_elong_right_vector.size(); ++j) {
139 arma::urowvec temp(leng_elong_left + v_dom_k_len + leng_elong_right_vector(j), arma::fill::ones);
140 temp.cols(leng_elong_left, leng_elong_left + v_dom_k.n_elem - 1) = v_dom_k;
141 v_dom_elong_left_right[k++] = temp;
142 }
143 }
144
145 // create the list containing all the possible v_dom_k elongated using compute_motif
146 const int v_elong_left_right_size = s_k_elong_left_right.size();
147 arma::uvec not_start_with_NA(v_elong_left_right_size,arma::fill::zeros);

149
150
151 for (int i = 0; i < v_elong_left_right_size; i++) {

153 if (std::holds_alternative<KMA::Mfield>(pair_motif_shift)){
154 v_elong_left_right.row(i) = *(std::get_if<KMA::Mfield>(&pair_motif_shift));
155 not_start_with_NA(i) = 1;
156 }
157 }
158
159 // filter centroid and shifts that are not in NA positions
160 auto not_NA_index = std::views::iota(0,v_elong_left_right_size)
161 | std::views::filter([&not_start_with_NA](int index_j){return(not_start_with_NA(index_j));});
162
163 KMA::Mfield filtered_v_elong(arma::accu(not_start_with_NA),Y.n_cols);
164 std::vector<arma::ivec> filtered_s_k(arma::accu(not_start_with_NA));
165
166 int i = 0;
167 for(const auto & index : not_NA_index){
168 filtered_v_elong.row(i) = v_elong_left_right.row(index);
169 filtered_s_k[i] = s_k_elong_left_right[index];
170 i = i + 1;
171 }
172
173 const unsigned int s_k_elong_left_right_size = filtered_s_k.size();
174
175 // compute performance index before elongation
176 double Jk_before = perf->compute_Jk(v_new_k,s_k,p_k,Y,param._w,param._m,
177 arma::datum::nan, KMA::uvector({}),
178 diss);
179
180 // compute performance indexes for all possible elongations
181 arma::vec Jk_after(s_k_elong_left_right_size);
182 double c_i;
183
184 for (arma::uword i = 0; i < s_k_elong_left_right_size; i++) {
185
186 c_i = std::max<double>(floor(util::findDomain(filtered_v_elong(i,0)).n_elem*(1 - param._max_gap)),c);
187
188 Jk_after[i] = perf->compute_Jk(filtered_v_elong.row(i),filtered_s_k[i],
189 p_k,Y,param._w,param._m,c_i,
190 keep_k,diss);
191 }
192
193 // find the best elongation in terms of perf. index
194 arma::vec diff_perc = (Jk_after-Jk_before)/Jk_before;
195 arma::uword best_elong = arma::index_min(diff_perc);
196
197 // check that the min really exists
198 bool elongate = false;
199 if (best_elong < diff_perc.size() and arma::conv_to<arma::Col<double>>::from(arma::find_nan(diff_perc)).size() != diff_perc.size())
200 elongate = diff_perc(best_elong) < param._deltaJK_elong;
201
202 // evaluate if elongate or not
203 if(elongate) {
204 V_new.row(index) = filtered_v_elong.row(best_elong);
205 V_dom[index] = v_dom_elong_left_right[best_elong + 1];
206 S_k.col(index) = filtered_s_k[best_elong]; // stampare questo
207 } else {
208 return;
209 }

212template<bool use1>
213void MotifSobol::elongate_motifs_helper(KMA::Mfield& V_new,
214 std::vector<arma::urowvec>& V_dom,
215 KMA::imatrix& S_k,const KMA::matrix& P_k,
216 const KMA::Mfield& Y,const KMA::matrix& D,
217 const Parameters& param,
218 const std::shared_ptr<PerformanceIndexAB>& perf,
219 const std::shared_ptr<Dissimilarity>& diss,
220 const Rcpp::Function & quantile_func) const

222#ifdef _OPENMP
223 // number of threads
224 const unsigned int n_threads = param._n_threads;
225#endif
226
227 std::size_t V_dom_size = V_dom.size();
228
229 arma::uvec len_dom(V_dom_size);
230 arma::uvec bool_gaps(V_dom_size);
231 for(unsigned int i = 0; i < V_dom_size; ++i){
232 len_dom(i) = V_dom[i].size();
233 bool_gaps(i) = (arma::accu(V_dom[i]) < len_dom(i));
234 }
235
236 arma::uvec with_gaps = arma::find(bool_gaps == 1);
237 unsigned int with_gaps_size = with_gaps.size();
238
239 // @TODO: check this part for domains with NaN
240 if (with_gaps_size != 0){
241
242 KMA::Mfield V_filled(with_gaps_size,Y.n_cols);
243
244 std::vector<KMA::uvector> V_dom_filled(with_gaps_size);
245
246 arma::vec Jk_before(with_gaps_size);
247
248 arma::vec Jk_after(with_gaps_size);
249
250 // fill the domains of the motifs with gaps and recompute the motifs with the filled domains
251 // and compute the perf.indexes before and after the filling
252 #ifdef _OPENMP
253 #pragma omp parallel for num_threads(n_threads)
254 #endif
255 for (unsigned int i = 0; i < with_gaps_size; ++i){
256
257 V_dom_filled[i] = arma::uvec(V_dom[with_gaps(i)].n_elem,arma::fill::ones);
258 const auto & variant_motif = compute_motif_helper<use1>(V_dom_filled[i], // errore
259 S_k.col(with_gaps(i)),
260 P_k.col(with_gaps(i)),
261 Y,param._m);
262
263 V_filled.row(i) = *(std::get_if<KMA::Mfield>(&variant_motif)); // Possibile errore se ritorna Nullptr
264
265 Jk_before(i) = perf->compute_Jk(V_new.row(with_gaps(i)),
266 S_k.col(with_gaps(i)),
267 P_k.col(with_gaps(i)),
268 Y,param._w,param._m,
269 arma::datum::nan,arma::uvec({}),
270 diss);
271
272 Jk_after(i) = perf->compute_Jk(V_filled.row(i),
273 S_k.col(with_gaps(i)),
274 P_k.col(with_gaps(i)),
275 Y,param._w,param._m,
276 arma::datum::nan,arma::uvec({}),
277 diss);
278 }
279
280 // if filling the domain improves the perf. index over a certain threshold replace the domain and the motifs with the filled one
281 const arma::uvec& fill = arma::find((Jk_after-Jk_before)/Jk_before < param._deltaJK_elong);
282 for(unsigned int i=0; i < fill.size(); ++i){
283 V_dom[with_gaps(i)] = V_dom_filled[i];
284 V_new.row(with_gaps(i)) = V_filled.row(i);
285 }
286 }
287
288 std::vector<arma::ivec> len_elong(V_dom_size);
289 for (unsigned int i = 0; i < V_dom_size; ++i){
290 const unsigned int len_max_elong_i = std::min<unsigned int>(std::floor(len_dom[i]*param._max_elong),
291 param._c_max[i] - len_dom[i]);
292 if (len_max_elong_i == 0){
293 len_elong[i] = KMA::ivector{};
294 }
295 else{
296 len_elong[i] = (len_max_elong_i <= param._trials_elong) ?
297 arma::regspace<arma::ivec>(1, len_max_elong_i):
298 round(arma::linspace<arma::ivec>(1, len_max_elong_i, param._trials_elong));
299 }
300 }
301
302 double quant = Rcpp::as<double>(quantile_func(D,0.25));
303 // keep will be a matrix whose value i,j will be D(i,j) < quantile(0)
304 arma::umat keep = D < quant;
305 // col-wise sum of the matrix keep
306 const arma::uvec& col_sum_keep = (sum(keep, 0)).t();
307 // vector of bool = true iff col_sum_keep[i]==0
308 const arma::uvec& col_sum_keep_zero = (col_sum_keep==0);
309 // empty_k stores the indexes of the col that have col_sum_keep = 0
310 const arma::uvec& empty_k = find(col_sum_keep_zero);
311
312 for (auto k : empty_k){
313 const unsigned int min_col_k_D = index_min(D.col(k));
314 keep(min_col_k_D, k) = true;
315 }
316
317 #ifdef _OPENMP
318 #pragma omp parallel for num_threads(n_threads)
319 #endif
320 for (unsigned int i = 0; i < V_dom_size; ++i){
321 elongation<use1>(V_new,V_dom,S_k,P_k.col(i),
322 len_elong[i],keep.col(i),param._c[i],
323 Y,i,param,perf,diss);
324 }
325}
